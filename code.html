/export class AddDocumentsComponent implements OnInit {
  addDocumentForm: FormGroup = new FormGroup({});

  // Full lists
  documentTypes: DocumentTypeResponse[] = [];
  documentTypesAndCategories: DocumentTypeResponse[] = [];

  // Filtered lists for dropdowns
  filteredDocumentTypes: DocumentTypeResponse[] = [];
  filteredDocumentTypesAndCategories: DocumentTypeResponse[] = [];

  classificationType: string = ''; // Ensure this is properly initialized

  ngOnInit() {
    // Initialize filtered lists with full lists
    this.filteredDocumentTypes = [...this.documentTypes];
    this.filteredDocumentTypesAndCategories = [...this.documentTypesAndCategories];
  }

  // Filters the first dropdown
  onTypeFilter(value: string): void {
    this.filteredDocumentTypes = this.documentTypes.filter((type) =>
      (type.name.toLowerCase().includes(value.toLowerCase()) ||
        type.categoryName?.toLowerCase().includes(value.toLowerCase())) &&
      type.classification === this.classificationType
    );
  }

  // Filters the second dropdown (fixed version)
  onCategoryFilter(value: string): void {
    this.filteredDocumentTypesAndCategories = this.documentTypesAndCategories.filter((category) =>
      (category.docTypeName.toLowerCase().includes(value.toLowerCase()) ||
        category.categoryName?.toLowerCase().includes(value.toLowerCase()))
    );
  }
}
